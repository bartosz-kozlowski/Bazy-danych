1.

SELECT p.NAZWISKO, z.ID_ZESP, z.NAZWA FROM PRACOWNICY p LEFT OUTER JOIN ZESPOLY z
 ON p.ID_ZESP = z.ID_ZESP
 ORDER BY p.NAZWISKO;

2.

SELECT z.NAZWA, z.ID_ZESP, CASE WHEN p.NAZWISKO IS NOT NULL THEN p.NAZWISKO
ELSE 'brak pracowników' END AS PRACOWNIK FROM ZESPOLY z LEFT OUTER JOIN PRACOWNICY P
ON z.ID_ZESP = p.ID_ZESP
ORDER BY NAZWA, NAZWISKO;

3.

SELECT CASE WHEN z.NAZWA IS NOT NULL THEN z.NAZWA ELSE 'brak zespolu' END AS ZESPOLY, CASE WHEN p.NAZWISKO IS NOT NULL THEN p.NAZWISKO ELSE 'brak pracowników' END AS PRACOWNIK FROM ZESPOLY z FULL OUTER JOIN PRACOWNICY p
ON z.ID_ZESP = p.ID_ZESP
ORDER BY NAZWA, NAZWISKO;

4.

SELECT z.NAZWA AS ZESPOL, COUNT(p.NAZWISKO) AS LICZBA, SUM(p.PLACA_POD) AS SUMA_PLAC FROM ZESPOLY z LEFT OUTER JOIN PRACOWNICY p
ON z.ID_ZESP = p.ID_ZESP
GROUP BY z.NAZWA
ORDER BY z.NAZWA;

5.

SELECT z.NAZWA AS NAZWA FROM ZESPOLY z LEFT OUTER JOIN PRACOWNICY p
ON z.ID_ZESP = p.ID_ZESP
MINUS
SELECT z.NAZWA AS NAZWA FROM ZESPOLY z RIGHT OUTER JOIN PRACOWNICY p
ON z.ID_ZESP = p.ID_ZESP
ORDER BY NAZWA;

6.

SELECT p.NAZWISKO AS PRACOWNIK, p.ID_PRAC, z.NAZWISKO AS SZEF, p.ID_SZEFA FROM PRACOWNICY p LEFT OUTER JOIN PRACOWNICY z
ON p.ID_SZEFA = z.ID_PRAC
ORDER BY p.NAZWISKO;

7.

SELECT p.NAZWISKO AS PRACOWNIK, COUNT(z.NAZWISKO) AS LICZBA_PODWLADNYCH
FROM PRACOWNICY p LEFT OUTER JOIN PRACOWNICY z
ON z.ID_SZEFA = p.ID_PRAC
GROUP BY p.NAZWISKO
ORDER BY p.NAZWISKO;

8.

SELECT p.NAZWISKO, p.ETAT, p.PLACA_POD, z.NAZWA, s.NAZWISKO AS SZEF FROM PRACOWNICY p
LEFT OUTER JOIN PRACOWNICY s
ON p.ID_SZEFA = s.ID_PRAC
LEFT OUTER JOIN ZESPOLY z
ON p.ID_ZESP = z.ID_ZESP
ORDER BY p.NAZWISKO;

9.

SELECT NAZWISKO, NAZWA FROM PRACOWNICY CROSS JOIN ZESPOLY
ORDER BY NAZWISKO, NAZWA;
10.
SELECT COUNT(*) FROM ETATY CROSS JOIN PRACOWNICY CROSS JOIN ZESPOLY;
11.
SELECT DISTINCT ETAT FROM PRACOWNICY WHERE EXTRACT(YEAR FROM ZATRUDNIONY) = 1992 
INTERSECT
SELECT DISTINCT ETAT FROM PRACOWNICY WHERE EXTRACT(YEAR FROM ZATRUDNIONY) = 1993
ORDER BY ETAT;
12.
SELECT ID_ZESP FROM ZESPOLY
EXCEPT
SELECT ID_ZESP FROM PRACOWNICY;
13.
SELECT z.ID_ZESP, z.NAZWA 
FROM ZESPOLY z
WHERE z.ID_ZESP IN (
    SELECT ID_ZESP FROM ZESPOLY
    EXCEPT
    SELECT ID_ZESP FROM PRACOWNICY
);
14.
SELECT NAZWISKO, PLACA_POD, 'Poniżej 480 złotych' AS PROG
FROM PRACOWNICY WHERE PLACA_POD < 480
UNION
SELECT NAZWISKO, PLACA_POD, 'Dokładnie 480 złotych' AS PROG
FROM PRACOWNICY WHERE PLACA_POD = 480
UNION
SELECT NAZWISKO, PLACA_POD, 'Powyżej 480 złotych' AS PROG
FROM PRACOWNICY WHERE PLACA_POD > 480
ORDER BY PLACA_POD, NAZWISKO;
