1. SELECT p.NAZWISKO, p.ETAT, p.ID_ZESP, z.NAZWA
FROM PRACOWNICY p INNER JOIN ZESPOLY z ON p.ID_ZESP = z.ID_ZESP
ORDER BY p.NAZWISKO;

2. SELECT p.NAZWISKO, p.ETAT, p.ID_ZESP, z.NAZWA
FROM PRACOWNICY p INNER JOIN ZESPOLY z ON p.ID_ZESP = z.ID_ZESP
WHERE z.ADRES = 'PIOTROWO 3A'
ORDER BY p.NAZWISKO;

3.

SELECT p.NAZWISKO, p.ETAT, p.PLACA_POD, e.PLACA_MIN, e.PLACA_MAX
FROM PRACOWNICY p INNER JOIN ETATY e ON p.ETAT = e.NAZWA
ORDER BY p.ETAT, p.NAZWISKO;

4.SELECT p.NAZWISKO, p.ETAT, p.PLACA_POD, e.PLACA_MIN, e.PLACA_MAX,
CASE WHEN p.PLACA_POD >= e.PLACA_MIN AND e.PLACA_MAX >= p.PLACA_POD THEN 'OK' ELSE 'NIE' END AS CZY_PENSJA_OK
FROM PRACOWNICY p INNER JOIN ETATY e ON p.ETAT = e.NAZWA
ORDER BY p.ETAT, p.NAZWISKO;

5.

SELECT p.NAZWISKO, p.ETAT, p.PLACA_POD, e.PLACA_MIN, e.PLACA_MAX,
CASE WHEN p.PLACA_POD >= e.PLACA_MIN AND e.PLACA_MAX >= p.PLACA_POD THEN 'OK' ELSE 'NIE' END AS CZY_PENSJA_OK
FROM PRACOWNICY p INNER JOIN ETATY e ON p.ETAT = e.NAZWA
WHERE p.PLACA_POD < e.PLACA_MIN OR e.PLACA_MAX < p.PLACA_POD
ORDER BY p.ETAT, p.NAZWISKO;

6.

SELECT p.NAZWISKO, p.PLACA_POD, p.ETAT, e.NAZWA AS KAT_PLAC, e.PLACA_MIN, e.PLACA_MAX
FROM PRACOWNICY p INNER JOIN ETATY e ON p.PLACA_POD >= e.PLACA_MIN AND e.PLACA_MAX >= p.PLACA_POD
ORDER BY p.NAZWISKO, KAT_PLAC;

7.

SELECT p.NAZWISKO, p.PLACA_POD, p.ETAT, e.NAZWA AS KAT_PLAC, e.PLACA_MIN, e.PLACA_MAX
FROM PRACOWNICY p INNER JOIN ETATY e ON p.PLACA_POD >= e.PLACA_MIN AND e.PLACA_MAX >= p.PLACA_POD
WHERE e.NAZWA = 'SEKRETARKA'
ORDER BY p.NAZWISKO;

8.
SELECT P.NAZWISKO AS PRACOWNIK, P.ID_PRAC, S.NAZWISKO AS SZEF, S.ID_PRAC AS ID_SZEFA
FROM PRACOWNICY P INNER JOIN PRACOWNICY S ON P.ID_SZEFA = S.ID_PRAC
ORDER BY P.NAZWISKO;

9.

SELECT p.NAZWISKO AS PRACOWNIK, p.ZATRUDNIONY AS PRAC_ZATRUDNIONY,
s.NAZWISKO AS SZEF, S.ZATRUDNIONY AS SZEF_ZATRUDNIONY, EXTRACT(YEAR FROM p.zatrudniony) - EXTRACT(YEAR FROM s.ZATRUDNIONY) AS LATA
FROM PRACOWNICY p INNER JOIN PRACOWNICY s ON p.ID_SZEFA = s.ID_PRAC
WHERE EXTRACT(YEAR FROM p.zatrudniony) - EXTRACT(YEAR FROM s.ZATRUDNIONY) <= 10
ORDER BY PRAC_ZATRUDNIONY, PRACOWNIK;

10. SELECT z.NAZWA, COUNT(*) AS LICZBA, AVG(p.PLACA_POD) AS SREDNIA_PLACA
FROM ZESPOLY z INNER JOIN PRACOWNICY p ON p.ID_ZESP = z.ID_ZESP
GROUP BY z.NAZWA
ORDER BY z.NAZWA;

11. SELECT z.NAZWA, CASE
WHEN COUNT(*) <= 2 THEN 'MAŁY'
WHEN COUNT(*) >=3 AND COUNT(*) <= 6 THEN 'ŚREDNI'
WHEN COUNT(*) >= 7 THEN 'DUŻY'
END AS ETYKIETA
FROM ZESPOLY z INNER JOIN PRACOWNICY p ON p.ID_ZESP = z.ID_ZESP
GROUP BY z.NAZWA
HAVING COUNT(*) > 0
ORDER BY z.NAZWA;
